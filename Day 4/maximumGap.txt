class Solution {
public:
    int maximumGap(vector<int>& nums) {
        int n=nums.size();
        sort(nums.begin(),nums.end());
        if(n==1){
            return 0;
        }
        if(n==2){
            return nums[1]-nums[0];
        }
        int x;
        x=nums[1]-nums[0];
        for (int i=1;i<n-1;i++){
            if((nums[i+1]-nums[i])>x){
                x=nums[i+1]-nums[i];
            }
        }
        return x;
    }
};
